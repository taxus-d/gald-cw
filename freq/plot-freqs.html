ratio_cache = set()
def comp_ratio(arr, *args):
    global ratio_cache
    for s in args:
        if s not in ratio_cache:
            ratio_cache = ratio_cache.union([s])
        r = s.split('/')
        if len(r) >= 2:
            arr[s] = arr[r[0]]/arr[r[1]]

d = dict(a=1,b=2)
comp_ratio(d,'a/b', 'b/a')
ratio_cache,d
